⚙ Google Sheets Setup Guide

This one-time setup takes about 10 minutes and makes client data live — clients will see real-time updates just by searching. Completely free, no hosting needed.

Step 1
Create a Google Sheet
Go to sheets.google.com and create a new blank spreadsheet. Name it "JVALU ICV Tracker". You can use your existing Google account.
Step 2
Open Apps Script Editor
In your Google Sheet, click Extensions → Apps Script. This opens the script editor. Delete any existing code and paste the entire script below:
// JVALU ICV Tracker — Google Apps Script Backend // Paste this ENTIRE block into Apps Script editor const SHEET_NAME = "Clients"; const DOCS_SHEET = "Documents"; function doGet(e) { const action = e.parameter.action; const ss = SpreadsheetApp.getActiveSpreadsheet(); if (action === "getAll") { try { const clients = getAllClients(ss); return jsonResponse({ success: true, clients: clients }); } catch(err) { return jsonResponse({ success: false, error: err.message }); } } return jsonResponse({ success: false, error: "Unknown action" }); } function doPost(e) { const body = JSON.parse(e.postData.contents); const action = body.action; const ss = SpreadsheetApp.getActiveSpreadsheet(); try { if (action === "saveClient") { saveClient(ss, body.client); return jsonResponse({ success: true }); } if (action === "saveDocs") { saveDocs(ss, body.clientId, body.documents); return jsonResponse({ success: true }); } if (action === "deleteClient") { deleteClient(ss, body.id); return jsonResponse({ success: true }); } return jsonResponse({ success: false, error: "Unknown action" }); } catch(err) { return jsonResponse({ success: false, error: err.message }); } } function jsonResponse(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function getOrCreateSheet(ss, name, headers) { let sh = ss.getSheetByName(name); if (!sh) { sh = ss.insertSheet(name); sh.getRange(1, 1, 1, headers.length).setValues([headers]); } return sh; } function getAllClients(ss) { const cSh = getOrCreateSheet(ss, SHEET_NAME, ["id","company","tradeLicense","contact","phone","email","financialYear","stage","addedDate","notes"]); const dSh = getOrCreateSheet(ss, DOCS_SHEET, ["clientId","docId","received","date","remarks"]); const cData = cSh.getDataRange().getValues(); const dData = dSh.getDataRange().getValues(); if (cData.length <= 1) return []; const cHeaders = cData[0]; const dHeaders = dData[0]; // Build docs map const docsMap = {}; for (let i = 1; i < dData.length; i++) { const row = dData[i]; const clientId = row[dHeaders.indexOf("clientId")]; const docId = parseInt(row[dHeaders.indexOf("docId")]); if (!docsMap[clientId]) docsMap[clientId] = {}; docsMap[clientId][docId] = { received: row[dHeaders.indexOf("received")] === true || row[dHeaders.indexOf("received")] === "TRUE", date: row[dHeaders.indexOf("date")] || "", remarks: row[dHeaders.indexOf("remarks")] || "" }; } const clients = []; for (let i = 1; i < cData.length; i++) { const row = cData[i]; if (!row[0]) continue; const client = {}; cHeaders.forEach((h, j) => { client[h] = row[j] || ""; }); client.documents = docsMap[client.id] || {}; clients.push(client); } return clients; } function saveClient(ss, client) { const sh = getOrCreateSheet(ss, SHEET_NAME, ["id","company","tradeLicense","contact","phone","email","financialYear","stage","addedDate","notes"]); const headers = sh.getRange(1, 1, 1, 10).getValues()[0]; const data = sh.getDataRange().getValues(); // Find existing row let rowIdx = -1; for (let i = 1; i < data.length; i++) { if (data[i][0] === client.id) { rowIdx = i + 1; break; } } const row = headers.map(h => client[h] !== undefined ? client[h] : ""); if (rowIdx > 0) { sh.getRange(rowIdx, 1, 1, row.length).setValues([row]); } else { sh.appendRow(row); } } function saveDocs(ss, clientId, documents) { const sh = getOrCreateSheet(ss, DOCS_SHEET, ["clientId","docId","received","date","remarks"]); const data = sh.getDataRange().getValues(); // Remove existing docs for this client for (let i = data.length - 1; i >= 1; i--) { if (data[i][0] === clientId) { sh.deleteRow(i + 1); } } // Append new docs Object.entries(documents).forEach(([docId, doc]) => { sh.appendRow([clientId, parseInt(docId), doc.received, doc.date || "", doc.remarks || ""]); }); } function deleteClient(ss, id) { const cSh = ss.getSheetByName(SHEET_NAME); const dSh = ss.getSheetByName(DOCS_SHEET); if (cSh) { const data = cSh.getDataRange().getValues(); for (let i = data.length - 1; i >= 1; i--) { if (data[i][0] === id) { cSh.deleteRow(i + 1); break; } } } if (dSh) { const data = dSh.getDataRange().getValues(); for (let i = data.length - 1; i >= 1; i--) { if (data[i][0] === id) { dSh.deleteRow(i + 1); } } } }
Step 3
Deploy as Web App
1. In the Apps Script editor, click Deploy → New Deployment
2. Click the gear icon ⚙ next to "Select type" → choose Web app
3. Set Execute as: Me
4. Set Who has access: Anyone
5. Click Deploy → Click Authorize access → Allow
6. Copy the Web app URL (looks like: https://script.google.com/macros/s/ABC.../exec)
